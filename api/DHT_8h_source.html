111 struct ComfortProfile
112 {
113  //Represent the 4 line equations:
114  //dry, humid, hot, cold, using the y = mx + b formula
115  float m_tooHot_m, m_tooHot_b;
116  float m_tooCold_m, m_tooHCold_b;
117  float m_tooDry_m, m_tooDry_b;
118  float m_tooHumid_m, m_tooHumid_b;
119 
120  inline bool isTooHot(float temp, float humidity)
121  {return (temp > (humidity * m_tooHot_m + m_tooHot_b));}
122  inline bool isTooHumid(float temp, float humidity)
123  {return (temp > (humidity * m_tooHumid_m + m_tooHumid_b));}
124  inline bool isTooCold(float temp, float humidity)
125  {return (temp < (humidity * m_tooCold_m + m_tooHCold_b));}
126  inline bool isTooDry(float temp, float humidity)
127  {return (temp < (humidity * m_tooDry_m + m_tooDry_b));}
128 
129  inline float distanceTooHot(float temp, float humidity)
130  {return temp - (humidity * m_tooHot_m + m_tooHot_b);}
131  inline float distanceTooHumid(float temp, float humidity)
132  {return temp - (humidity * m_tooHumid_m + m_tooHumid_b);}
133  inline float distanceTooCold(float temp, float humidity)
134  {return (humidity * m_tooCold_m + m_tooHCold_b) - temp;}
135  inline float distanceTooDry(float temp, float humidity)
136  {return (humidity * m_tooDry_m + m_tooDry_b) - temp;}
137 };
